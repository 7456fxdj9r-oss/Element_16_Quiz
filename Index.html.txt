<script>
    // Questions from the document [cite: 12-54]
    const questions = [
        "1. I find it difficult to control how much I eat",
        "2. I crash in the afternoon and have trouble waking up in the morning",
        "3. I have cravings for specific foods",
        "4. I suffer from moodiness, headaches, or fogginess",
        "5. When I only eat a moderate amount of food I am not satisfied",
        "6. I feel guilty after eating",
        "7. I spend a considerable amount of time thinking about food",
        "8. I eat even if I am not hungry because I crave a food",
        "9. I eat large amounts of food and feel powerless to stop",
        "10. I crave carbohydrates like pasta, bread, white rice, or desserts"
    ];

    const options = ["Rarely", "Occasionally", "Daily"];

    // Render the questions
    const quizContent = document.getElementById('quiz-content');

    questions.forEach((q, index) => {
        const item = document.createElement('div');
        item.className = 'question-item';

        const qText = document.createElement('span');
        qText.className = 'question-text';
        qText.textContent = q;
        item.appendChild(qText);

        const optionsDiv = document.createElement('div');
        optionsDiv.className = 'options-wrapper';

        options.forEach(opt => {
            const label = document.createElement('label');
            label.className = 'radio-label';
            
            const input = document.createElement('input');
            input.type = 'radio';
            input.name = `q${index}`;
            input.value = opt;
            
            const checkmark = document.createElement('span');
            checkmark.className = 'checkmark';

            label.appendChild(input);
            label.appendChild(checkmark);
            label.appendChild(document.createTextNode(opt));
            
            optionsDiv.appendChild(label);
        });

        item.appendChild(optionsDiv);
        quizContent.appendChild(item);
    });

    // Scoring Logic Source: 
    function calculateResults() {
        const form = document.getElementById('quizForm');
        let counts = { "Rarely": 0, "Occasionally": 0, "Daily": 0 };
        let answered = 0;

        for (let i = 0; i < questions.length; i++) {
            const radios = form.elements[`q${i}`];
            for (const radio of radios) {
                if (radio.checked) {
                    counts[radio.value]++;
                    answered++;
                    break;
                }
            }
        }

        if (answered < questions.length) {
            alert("Please answer all questions to see your profile.");
            return;
        }

        const resultArea = document.getElementById('result-area');
        let title, msg, className;

        // Logic: Greater than 6 in a specific category
        if (counts["Daily"] > 6) {
            title = "High Susceptibility";
            msg = "Your results indicate a high level of risk for food addiction cues.";
            className = "res-high";
        } else if (counts["Occasionally"] > 6) {
            title = "Moderate Susceptibility";
            msg = "Your results indicate a moderate level of risk.";
            className = "res-mod";
        } else if (counts["Rarely"] > 6) {
            title = "Low Susceptibility";
            msg = "Your results indicate a low level of risk.";
            className = "res-low";
        } else {
            title = "Mixed Susceptibility";
            msg = "Your answers are mixed. You do not show a dominant pattern (>6) in any single category.";
            className = "res-mixed";
        }

        resultArea.innerHTML = `
            <h2>${title}</h2>
            <p>${msg}</p>
            <div class="score-badge">
                Rarely: ${counts["Rarely"]} | Occasionally: ${counts["Occasionally"]} | Daily: ${counts["Daily"]}
            </div>
        `;
        
        resultArea.className = className;
        resultArea.style.display = 'block';
        resultArea.scrollIntoView({ behavior: 'smooth' });
    }
</script>
